func sleep() {
    time.Sleep(1 * time.Millisecond)
}

func BenchmarkSleepParallel(b *testing.B) {
    for i := 1; i <= 8; i *= 2 {
        b.Run(strconv.Itoa(i), func(b *testing.B) {
            b.SetParallelism(i)
            b.RunParallel(func(pb *testing.PB) {
                for pb.Next() {
                    sleep()
                }
            })
        })
    }
}
